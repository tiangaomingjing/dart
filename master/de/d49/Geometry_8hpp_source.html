471 template <typename MatrixType, typename ReturnType>
472 void extractNullSpace(const Eigen::JacobiSVD<MatrixType>& _SVD, ReturnType& _NS)
473 {
474 #if EIGEN_VERSION_AT_LEAST(3,2,2)
475  const int rank = _SVD.rank();
476 #else
477  int rank = 0;
478  if(_SVD.nonzeroSingularValues() > 0)
479  {
480  double thresh = std::max(_SVD.singularValues().coeff(0)*1e-10,
481  std::numeric_limits<double>::min());
482  int i = _SVD.nonzeroSingularValues()-1;
483  while( i >= 0 && _SVD.singularValues().coeff(i) < thresh )
484  --i;
485  rank = i+1;
486  }
487 #endif
488 
489  const int cols = _SVD.matrixV().cols();
490  const int rows = _SVD.matrixV().rows();
491  _NS = _SVD.matrixV().block(0, rank, rows, cols-rank);
492 }

587  BoundingBox();
588  BoundingBox(const Eigen::Vector3d& min, const Eigen::Vector3d& max);
589 
590  inline const Eigen::Vector3d& getMin() const { return mMin; }

592 
593  inline void setMin(const Eigen::Vector3d& min) { mMin = min; }
594  inline void setMax(const Eigen::Vector3d& max) { mMax = max; }
595 
596  // \brief Centroid of the bounding box (i.e average of min and max)
597  inline Eigen::Vector3d computeCenter() const { return (mMax + mMin) * 0.5; }
598  // \brief Coordinates of the maximum corner with respect to the centroid.

600  // \brief Length of each of the sides of the bounding box.
601  inline Eigen::Vector3d computeFullExtents() const { return (mMax - mMin); }
602 
603  protected:
604  // \brief minimum coordinates of the bounding box
605  Eigen::Vector3d mMin;
606  // \brief maximum coordinates of the bounding box
607  Eigen::Vector3d mMax;
